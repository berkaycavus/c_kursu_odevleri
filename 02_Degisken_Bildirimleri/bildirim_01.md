#### Aşağıdaki bildirimlerden geçersiz olanları işaretleyiniz: 

```
int main()
{
	signed x1;		-->	true
	unsigned double x2;	-->	false
	long int long x3;	-->	true
	bool x4;		-->	false
	char int x5;		-->	false
	x6 float;		-->	false
	long double x7;		-->	true
	unsigned short int x8;	-->	true
	long unsigned x9;	-->	true
	long float x10;		-->	true
}
```
